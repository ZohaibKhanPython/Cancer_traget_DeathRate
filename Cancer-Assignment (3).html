Now that we have the split function KFold , i will be using the function to split the data in to X and Y train and test and then pipeline is used to assemble several steps that can be cross-validated together while setting different parameters. and then working out the score to calculate the accuracy of data

In [9]:
##import numpy as np

# Empty vectors of scores
R2_train = []
R2_test = []

for train_index, test_index in kf.split(X):
    X_train, X_test = X.iloc[train_index,:], X.iloc[test_index, :]
    y_train, y_test = y.iloc[train_index,:], y.iloc[test_index, :]
    ##building a pipeline
    pipe_deg1 = Pipeline([('scaler', StandardScaler()),
                         ('poly', PolynomialFeatures(1)), ##1 is producing better results
                          ('lm', LinearRegression())]) # training a linear regression model    
    
    # using the pipeline
    pipe_deg1.fit(X_train, y_train)
    
    R2_train.append( pipe_deg1.score(X_train, y_train) )
    R2_test.append( pipe_deg1.score(X_test, y_test) )

    # Use the trained model to create predictions on train and test data
    y_pred = pipe_deg1.predict(X_test)
    y_train_pred = pipe_deg1.predict(X_train)

R2_train_mean = round(np.mean(R2_train), 2)
R2_test_mean = round(np.mean(R2_test), 2)

print("Mean R2 on train:", R2_train_mean )
print("Mean R2 on test:", R2_test_mean )
pipe_deg1.score(X_test,y_test)
